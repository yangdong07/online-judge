#### Analysis

这题主要考察 递归和字符串操作。

C++ 的拼接字符串还是挺麻烦的， `char *` 不可以直接拼接， 必须转成 string 才可以拼。

至于递归，仔细观察一下就可以了。


```cpp
#include <iostream>
#include <string>

using namespace std;


string expression(int n)
{
    string s = "";
    if (n & 1) s = string("+2(0)") + s;
    if (n & 2) s = string("+2") + s;

    for (int i = 4, c = 2; i <= n; i <<= 1, ++c)
    {
        if (n & i) s = string("+2(") + expression(c) + string(")") + s;
    }
    return s.substr(1);
}

int main()
{
    int n;
    cin >> n;
    cout << expression(n) << endl;

}
```
